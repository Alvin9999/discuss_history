#5 GoAgentDemoNew 
#1 补充
VPS装debian的话，非常方便
apt-get install youtube-dl apache2
cd /var/www/html
rm -rf *.*
##下载单个视频
youtube-dl --format best --audio-format best https://www.youtube.com/watch?v=QcIy9NiNbmo
##下载专辑
youtube-dl --format best --audio-format best https://www.youtube.com/playlist?list=UUANLZYMidaCbLQFWXBC95J
##下载某用户上传的所有视频
youtube-dl --format best --audio-format best https://www.youtube.com/user/clubwubai
##下载某用户上传的含某关键字的视频
youtube-dl --format best --audio-format best https://youtube.com/user/universaltwn --match-title 伍佰

然后http://VPSIP 就可以下载了
