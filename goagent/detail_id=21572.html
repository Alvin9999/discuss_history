#13 新建记事本，复制下方代码，另存为 bandip.py，放到 checkgoogleip 目录下，然后运行 python bandip.py 即可生成一个 ip_bind.txt，而且ip延时都是1000以内，延时多少可以自行更改。

iplist = []
with open('ip_tmpok.txt') as ip_tmpok:
    for x in ip_tmpok:
        sline = x.strip().split(' ')
        iplist.append(sline)
with open('ip_bind.txt', 'w') as ip_bind:
    iplist.sort(key=lambda x: int(x[1]))
    out = '|'.join(x[0] for x in iplist if int(x[1]) < 1000)
    ip_bind.write(out)
