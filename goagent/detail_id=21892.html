# 监听地址，设为0.0.0.0可以监听所有端口，共享给局域网使用
listen = http://127.0.0.1:4411
# 至少指定一个上级代理
# SOCKS5 二级代理
proxy = socks5://127.0.0.1:1080
# HTTP 二级代理
proxy = http://127.0.0.1:8087
# shadowsocks 二级代理
proxy = ss://aes-128-cfb:password@example.server.com:25

#44 redfrog...@gmail.com 
#40 用rc-full.txt 这个文档，让后把rc.txt改名或者删掉，再把rc-full.txt 改成rc.txt，也怪我没给你们说清楚，作者预留的rc.txt 就是个坑爹货。然后按照我的范本来修改这几处吧：

1.这里这么改，要去掉前面的#号，下边都一样
#############################
# 通过IP判断是否直连
#############################
judgeByIP = false

2.这里这么改
# 指定多个二级代理时使用的负载均衡策略，可选策略如下
#
#   backup:  默认策略，优先使用第一个指定的二级代理，其他仅作备份使用
#   hash:    根据请求的 host name，优先使用 hash 到的某一个二级代理
#   latency: 优先选择连接延迟最低的二级代理
#
# 一个二级代理连接失败后会依次尝试其他二级代理
# 失败的二级代理会以一定的概率再次尝试使用，因此恢复后会重新启用
loadBalance = backup

其他配置都不要动，尤其是loadBalance千万别动，动了就报错。

#50 俺用 Proxifier GoAgent_3.23 使用 Psiphon, Lantern, Shadowsocks 作前置代理统统没问题。

猜测你会出错的原因： 

1. proxy.ini 和 proxy.user.ini 中 [proxy] 没有关闭，要关闭才行，操作如下：

[proxy]

autodetect = 1
host = 10.64.1.63
port = 8080
username =
password =

2. proxy.ini 和 proxy.user.ini 中你自定义的 IP 不太好用，我的自动义的 IP 

google_cn = 139.175.107.216|139.175.107.217|139.175.107.221|139.175.107.227|139.175.107.232|139.175.107.237|139.175.107.242|139.175.107.247
google_hk = 139.175.107.216|139.175.107.217|139.175.107.221|139.175.107.227|139.175.107.232|139.175.107.237|139.175.107.242|139.175.107.247

4. 你使用的 Proxifier 不是安装版，或你的 Proxifier 是安装版，但失效了，需重装

5. 或许你的 Shadowsocks 账号有问题，换个账号或换用赛风 / 蓝灯作 GoAgent 的前置代理

#170 你有两种选择，任选其一：

方案一： 同时开启 ss-qt5 和 meow. 在 meow 里写规则 proxy = socks5://127.0.0.1:1080 , 然后想办法让 GoAgent 使用 meow 的 http 代理 127.0.0.1:4411

方案二： 只开启 ss-qt5 （meow 开不开无所谓，因为用不到她），然后想办法让 GoAgent 使用 ss-qt5 的 socks5 代理 127.0.0.1:1080

“想办法让 GoAgent 使用”前置代理的方法有两类（这些方法近期讨论的帖子里有一大推，聪明的俺一定能找到），任选其一：

二选一之方案一： 在 GoAgent proxy.ini 中指定 GoAgent 使用前置代理（如果使用上面所说的方案一 （用了 meow ）就填 meow 的 127.0.0.1:441, 类型 http; 如果使用上面方案二就填 ss-qt5 的 127.0.0.1:1080, 类型 socks5)

二选一之方案二： 用 Proxifier 定义规则使 GoAgent 使用前置代理（如果使用上面所说的方案一 （用了 meow ）就填 meow 的 127.0.0.1:441, 类型 http; 如果使用上面方案二就填 ss-qt5 的 127.0.0.1:1080, 类型 socks5)
