devcodewak commented Aug 11, 2016 • edited

勤奋的作者yinqiwen今晚大量更新，修正win编译问题、加http前置、加paas的fakeSni指定等
也放出了官方三json文件。

以下简化配置仅用于测试 heroku paas wss

服务端：
server.json

{
    "Log": ["stdout"],
    "Encrypt":{
        "Method": "salsa20",
        "Key": "12345678901234567890123456789012"
    }
}

Procfile
web: gsnova

客户端:
client.json

{
    // "Log": ["stderr","gsnova.log"],
    "Log": ["stderr"],

    "Encrypt":{
        "Method": "salsa20",
        //key换为你自己的，长度随意，注意与服务端server.json中相同
        "Key": "12345678901234567890123456789012"
    },

    "Proxy":[
        //单个端口已可同时支持http和socks5,下面测试需要起了两个端口,可删
        {
            "Local": "127.0.0.1:1080",
            "PAC": [
                {
                "Remote": "PAAS"
                }
            ]
        },
        {
            "Local": "127.0.0.1:8787",
            "PAC": [
                {
                "Remote": "PAAS"
                }
            ]
        }
    ],

    "PAAS":{
        "Enable":true,
        "ServerList":[
            //实测用多sni ip配置情况下，配多app更快, myapp名改为你的
            "wss://myapp1.herokuapp.com",
            "wss://myapp2.herokuapp.com",
            "wss://myapp3.herokuapp.com",
            "wss://myapp4.herokuapp.com",
            "wss://myapp5.herokuapp.com"
        ],

        //上面多app情况下，这里直接配1，表示单app的连接数
        "ConnsPerServer": 1,
               //sni 中转ip组，注意在hosts.json中对应配置
        "SNIProxy" : "cn_sni_proxy"

        // 直连不中转
        //"SNIProxy": ""


    }

}

hosts.json

{
        //sni ip组，client.json中，最后一行 "SNIProxy" 指向此处，应选择合适的sni组才能稳定连接
    "cn_sni_proxy" :["120.198.243.3","120.198.243.115","120.198.243.113","120.198.243.52"]
}

采用官方配置中PAAS及cn_sni_proxy部分，现访问流畅多了。
网页不会再卡了，秒开。
